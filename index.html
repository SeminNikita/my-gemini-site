async function sendMessage() { const input = document.getElementById('user-input'); const chatWindow = document.getElementById('chat-window'); const userInput = input.value.trim(); if (!userInput) return; chatWindow.innerHTML += `
${userInput}
`; input.value = ''; try { // Мы используем относительный путь const response = await fetch('/api/chat', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ prompt: userInput }) }); // Если сервер ответил, но там ошибка (например, 404 или 500) if (!response.ok) { const errorData = await response.json().catch(() => ({ text: "Непонятная ошибка сервера" })); throw new Error(errorData.text || `Статус сервера: ${response.status}`); } const data = await response.json(); chatWindow.innerHTML += `
${data.text}
`; } catch (error) { console.error("Ошибка запроса:", error); // Выводим ТОЧНЫЙ текст ошибки в чат, чтобы мы его увидели chatWindow.innerHTML += `
Детали ошибки: ${error.message}
`; } chatWindow.scrollTop = chatWindow.scrollHeight; }
