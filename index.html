async function sendMessage() { const input = document.getElementById('user-input'); const chatWindow = document.getElementById('chat-window'); const userInput = input.value.trim(); if (!userInput) return; // Добавляем сообщение пользователя chatWindow.innerHTML += `
${userInput}
`; input.value = ''; try { // ВАЖНО: адрес должен начинаться со слэша / const response = await fetch('/api/chat', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ prompt: userInput }) }); const data = await response.json(); if (response.ok) { // Отображаем ответ ИИ chatWindow.innerHTML += `
${data.text}
`; } else { chatWindow.innerHTML += `
Ошибка: ${data.text || 'Неизвестная ошибка'}
`; } } catch (error) { console.error("Fetch error:", error); chatWindow.innerHTML += `
Ошибка: Не удалось связаться с сервером.
`; } chatWindow.scrollTop = chatWindow.scrollHeight; }
